# crop original image, only keep middle left area
def crop_img(img_path):
    im = Image.open(img_path)
    width, height = im.size
    
    left = 0
    top = height / 4
    right = width / 2
    bottom = 3 * height / 4
    
    result = im.crop((left, top, right, bottom))
    return result
